\begin{tikzpicture}[remember picture, overlay]
    % Draw the C-shaped swap arrow
    \draw[<->, thick, shorten <=2pt, shorten >=2pt]
        ([xshift=1em]row2.east) -- ++(1em,0)
        -- ([xshift=2em]row3.east) -- ([xshift=1em]row3.east);
\end{tikzpicture}

\end{array}
\right)
